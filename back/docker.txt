
  barber-api:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: barber-api
    ports:
      - '3333:3333'
    environment:
      - NODE_ENV=production
      - DATABASE_URL=postgresql://${POSTGRESQL_USERNAME}:${POSTGRESQL_PASSWORD}@barber-db:5432/${POSTGRESQL_DATABASE}
      - JWT_SECRET=${JWT_SECRET}
      # Configurações Datadog
      - DD_AGENT_HOST=datadog-agent
      - DD_TRACE_AGENT_PORT=8126
      - DD_ENV=production
      - DD_SERVICE=barbearia-api
      - DD_VERSION=1.0.0
      - DD_LOGS_INJECTION=true
    depends_on:
      - barber-db
      - datadog-agent
    labels:
      - 'com.datadoghq.ad.logs=[{"source":"nodejs","service":"barbearia-api"}]'
    restart: unless-stopped
    networks:
      - barber-network
    healthcheck:
      # Verifica se o serviço está respondendo na porta 3333
      test:
        [
          'CMD',
          'node',
          '-e',
          "require('http').get('http://localhost:3333/health', (r) => {process.exit(r.statusCode === 200 ? 0 : 1)})",
        ]
      interval: 30s
      timeout: 10s # Aumentado um pouco para evitar falso negativo
      retries: 3
      start_period: 40s

networks:
  barber-network:
    driver: bridge
